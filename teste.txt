# 1. Carregar pacotes
library(tidyverse)
library(tseries)
library(vars)
library(lubridate)

# 2. Carregar e preparar os dados
df <- read.csv2("dy_ibov.csv", stringsAsFactors = FALSE) %>%
  mutate(
    DATE = dmy(DATE),
    IBOV = as.numeric(gsub(",", ".", IBOV)),
    DY_IBOV = as.numeric(gsub(",", ".", DY_IBOV))
  ) %>%
  arrange(DATE)

# 3. Criar proxy de crescimento dos dividendos e retornos do IBOV
df_transformado <- df %>%
  mutate(
    # Proxy para o fluxo nominal de dividendos
    proxy_dividendos_nominais = IBOV * DY_IBOV,
    
    # Variação logarítmica (crescimento) da proxy de dividendos
    crescimento_dividendos = c(NA, diff(log(proxy_dividendos_nominais))),
    
    # Retorno logarítmico do IBOV (mesmo de antes)
    retorno_IBOV = c(NA, diff(log(IBOV)))
  ) %>%
  # Remove as linhas com NA geradas pelos cálculos de diff()
  na.omit()

# 4. Testar estacionariedade das novas séries
print("--- Teste ADF para retorno_IBOV ---")
adf.test(df_transformado$retorno_IBOV)

print("--- Teste ADF para crescimento_dividendos ---")
adf.test(df_transformado$crescimento_dividendos)

# 5. Selecionar lag ótimo e rodar o modelo VAR
ts_data_final <- ts(df_transformado[, c("retorno_IBOV", "crescimento_dividendos")], frequency = 12)

print("--- Seleção de Lags (p) para o Modelo Final ---")
lag_select <- VARselect(ts_data_final, lag.max = 12, type = "const")
print(lag_select$selection)

p_otimo <- lag_select$selection["AIC(n)"]
print(paste("Lag ótimo escolhido (AIC):", p_otimo))
var_model_final <- VAR(ts_data_final, p = p_otimo, type = "const")

# 6. Testar Causalidade de Granger com as novas variáveis
print("--- Causalidade: crescimento_dividendos -> retorno_IBOV? ---")
print(causality(var_model_final, cause = "crescimento_dividendos"))

print("--- Causalidade: retorno_IBOV -> crescimento_dividendos? ---")
print(causality(var_model_final, cause = "retorno_IBOV"))

# 7. Gerar e plotar as novas Funções de Resposta ao Impulso (IRF)
irf_cresc_div <- irf(var_model_final, impulse = "crescimento_dividendos", response = "retorno_IBOV", n.ahead = 24, boot = TRUE)
irf_ret_ibov <- irf(var_model_final, impulse = "retorno_IBOV", response = "crescimento_dividendos", n.ahead = 24, boot = TRUE)

plot(irf_cresc_div, main = "Resposta do Retorno IBOV a um choque no Crescimento dos Dividendos")
plot(irf_ret_ibov, main = "Resposta do Crescimento dos Dividendos a um choque no Retorno IBOV")
